    <script>
        // 轮播图功能
        let currentSlide = 0;
        const slides = document.querySelectorAll('.carousel-item');
        
        function showSlide(n) {
            slides.forEach(slide => slide.classList.remove('active'));
            slides[n].classList.add('active');
        }

        setInterval(() => {
            currentSlide = (currentSlide + 1) % slides.length;
            showSlide(currentSlide);
        }, 5000);

        // 图片放大功能
        document.querySelectorAll('.research-item img').forEach(img => {
            img.onclick = () => {
                const overlay = document.createElement('div');
                overlay.style.cssText = `
                    position: fixed;
                    top:0; left:0;
                    width:100vw; height:100vh;
                    background: rgba(0,0,0,0.8);
                    cursor: zoom-out;
                    display: grid;
                    place-items: center;
                    z-index: 999;
                `;
                overlay.innerHTML = `<img src="${img.src}" style="max-width:90%; max-height:90%;">`;
                overlay.onclick = () => overlay.remove();
                document.body.appendChild(overlay);
            }
        });

        // 团队卡片滑动功能
        const memberContainer = document.querySelector('.member-container');
        let isDown = false;
        let startX;
        let scrollLeft;

        memberContainer.addEventListener('mousedown', (e) => {
            isDown = true;
            startX = e.pageX - memberContainer.offsetLeft;
            scrollLeft = memberContainer.scrollLeft;
        });

        memberContainer.addEventListener('mouseleave', () => {
            isDown = false;
        });

        memberContainer.addEventListener('mouseup', () => {
            isDown = false;
        });

        memberContainer.addEventListener('mousemove', (e) => {
            if (!isDown) return;
            e.preventDefault();
            const x = e.pageX - memberContainer.offsetLeft;
            const walk = (x - startX) * 2;
            memberContainer.scrollLeft = scrollLeft - walk;
        });

        // 高德地图初始化
        function initMap() {
            var map = new AMap.Map('map-container', {
                zoom: 17,
                center: [114.367785, 30.539472],
                doubleClickZoom: false
            });
            
            var marker = new AMap.Marker({
                position: [114.35,30.55],
                title: "武汉大学医学研究院",
                label: {
                    content: "蒋白山课题组",
                    offset: new AMap.Pixel(20, -10)
                }
            });
            map.add(marker);
            
            map.plugin('AMap.Scale', function(){
                new AMap.Scale();
            });
        }

        // 平滑滚动导航
        document.querySelectorAll('.navbar a').forEach(anchor => {
            anchor.addEventListener('click', function(e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                target.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
            });
        });

        document.addEventListener('DOMContentLoaded', initMap);
    </script>
